=======                             Includes                             =======

#include "KeyManager.h"

/*==============================================================================
=======               Defines & Macros for General Purpose               =======
==============================================================================*/

/*==============================================================================
=======                        Constants & Types                         =======
==============================================================================*/

/*==============================================================================
=======                        Global variables                          =======
==============================================================================*/

/*==============================================================================
=======                        Local variables                           =======
==============================================================================*/

/*==============================================================================
=======                        Global Function                           =======
==============================================================================*/

/*==============================================================================
=======                        Local Function                            =======
==============================================================================*/
static void KeyManager_IdleState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode);
static void KeyManager_PressedState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode);
static void KeyManager_ShortPressState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode);
static void KeyManager_LongPressState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode);
static void KeyManager_StickyPressState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode);
static void KeyManager_Transit(KeyButton_t *ptKey, KeyStatus_e eNextState, const KeyStateHandler_pt ptNextHandler, KeyStatusSwitch_e eEvent);
static bool KeyManager_IsPressed(const KeyButton_t *ptKey);
static void KeyManager_UpdateTimers(KeyButton_t *ptKey, bool bIsPressed);
static void KeyManager_ResetKey(KeyButton_t *ptKey);

/**
 * @brief 状态跳转处理
 */
static void KeyManager_Transit(KeyButton_t *ptKey, KeyStatus_e eNextState, const KeyStateHandler_pt ptNextHandler, KeyStatusSwitch_e eEvent)
{
    if (NULL != ptKey)
    {
        ptKey->eState       = eNextState;
        ptKey->StateHandler = ptNextHandler;
        ptKey->eStatusEvent = eEvent;
    }
}

/**
 * @brief 判断物理层是否按下
 */
static bool KeyManager_IsPressed(const KeyButton_t *ptKey)
{
    bool     bResult         = false;
    uint16_t usPhysicalValue = 0U;

    if((NULL != ptKey) && (NULL != ptKey->fnReadCollectValue))
    {
        usPhysicalValue = ptKey->fnReadCollectValue();

        if (eKeyType_IO == ptKey->eType)
        {
            bResult = (eIoActiveLow == ptKey->uConfig.eIoActiveLevel) ? (0U == usPhysicalValue) : (0U != usPhysicalValue);
        }
        else
        {
            bResult = ((usPhysicalValue >= ptKey->uConfig.tAdcRange.usAdcMin) && (usPhysicalValue <= ptKey->uConfig.tAdcRange.usAdcMax));
        }
    }
   
    return bResult;
}

/**
 * @brief 更新时间计数器
 */
static void KeyManager_UpdateTimers(KeyButton_t *ptKey, bool bIsPressed)
{
    if(NULL != ptKey)
    {
        if (true == bIsPressed)
        {
            if (ptKey->usPressCount < UINT16_MAX)
            {
                ptKey->usPressCount++;
            }
            ptKey->usReleaseCount = 0U;
        }
        else
        {
            if (ptKey->usReleaseCount < UINT16_MAX)
            {
                ptKey->usReleaseCount++;
            }
            ptKey->usPressCount = 0U;
        }
    }
}

/**
 * @brief 强制重置按键状态
 */
static void KeyManager_ResetKey(KeyButton_t *ptKey)
{
    if(NULL != ptKey)
    {
        if (eKeyIdle != ptKey->eState)
        {
            KeyManager_Transit(ptKey, eKeyIdle, &KeyManager_IdleState, eAnyToIdle);
        }

        ptKey->usPressCount   = 0U;
        ptKey->usReleaseCount = 0U;
    }
}

static void KeyManager_IdleState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode)
{
    (void)eKeyTriggerMode;
    if (true == bIsPressed)
    {
        KeyManager_Transit(ptKey, eKeyPressed, &KeyManager_PressedState, eKeep);
    }
}

static void KeyManager_PressedState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode)
{
    KeyStatus_e       eNext = eKeyPressed;
    KeyStatusSwitch_e eEvt  = eKeep;

    if (NULL != ptKey)
    {
        if ((false == bIsPressed) && (ptKey->usReleaseCount >= ptKey->usReleaseThreshold))
        {
            KeyManager_Transit(ptKey, eKeyIdle, KeyManager_IdleState, eAnyToIdle);
        }
        else if (ptKey->usPressCount >= ptKey->usShortPressCount)
        {
            eNext = (eTrigger_Press == eKeyTriggerMode) ? eKeyShortPress : eKeyPressed;
            eEvt  = (eTrigger_Press == eKeyTriggerMode) ? eIdleToShort : eKeep;

            KeyManager_Transit(ptKey, eKeyShortPress, KeyManager_ShortPressState, eEvt);
            ptKey->eState = eNext; /* 覆盖 Transit 的赋值，确保松开模式下业务层不误判 */
        }
        else
        { 
            /* Keep */
        }
    }
}

static void KeyManager_ShortPressState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode)
{
    KeyStatus_e       eFinal = eKeyPressed;
    KeyStatus_e       eNext  = eKeyPressed;
    KeyStatusSwitch_e eEvt   = eKeep;
    
    if (NULL != ptKey)
    {
        if ((false == bIsPressed) && (ptKey->usReleaseCount >= ptKey->usReleaseThreshold))
        {
            eFinal = (eTrigger_Release == eKeyTriggerMode) ? eKeyShortPress : eKeyIdle;
            eEvt   = (eTrigger_Release == eKeyTriggerMode) ? eIdleToShort : eAnyToIdle;

            KeyManager_Transit(ptKey, eKeyIdle, KeyManager_IdleState, eEvt);
            ptKey->eState = eFinal; /* 这里是为了给业务层看，不参与实际状态切换 */
        }
        else if (ptKey->usPressCount >= ptKey->usLongPressCount)
        {
            eNext = (eTrigger_Press == eKeyTriggerMode) ? eKeyLongPress : eKeyPressed;
            eEvt  = (eTrigger_Press == eKeyTriggerMode) ? eShortToLong : eKeep;

            KeyManager_Transit(ptKey, eKeyLongPress, KeyManager_LongPressState, eEvt);
            ptKey->eState = eNext;
        }
        else
        { 
            /* Keep */
        }
    }
}

static void KeyManager_LongPressState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode)
{
    KeyStatus_e       eFinal = eKeyPressed;
    KeyStatus_e       eNext  = eKeyPressed;
    KeyStatusSwitch_e eEvt   = eKeep;

    if (NULL != ptKey)
    {
        if ((false == bIsPressed) && (ptKey->usReleaseCount >= ptKey->usReleaseThreshold))
        {
            eFinal = (eTrigger_Release == eKeyTriggerMode) ? eKeyLongPress : eKeyIdle;
            eEvt   = (eTrigger_Release == eKeyTriggerMode) ? eShortToLong : eAnyToIdle;

            KeyManager_Transit(ptKey, eKeyIdle, &KeyManager_IdleState, eEvt);
            ptKey->eState = eFinal;
        }
        else if (ptKey->usPressCount >= ptKey->usStickyCount)
        {
            eNext = (eTrigger_Press == eKeyTriggerMode) ? eKeyStickyPress : eKeyPressed;
            eEvt  = (eTrigger_Press == eKeyTriggerMode) ? eLongToSticky : eKeep;

            KeyManager_Transit(ptKey, eKeyStickyPress, &KeyManager_StickyPressState, eEvt);
            ptKey->eState = eNext;
        }
        else
        { 
            /* Keep */
        }
    }
}

static void KeyManager_StickyPressState(KeyButton_t *ptKey, bool bIsPressed, KeyTriggerMode_e eKeyTriggerMode)
{
    KeyStatus_e       eFinal = eKeyPressed;
    KeyStatusSwitch_e eEvt   = eKeep;
    
    if (NULL != ptKey)
    {
        if ((false == bIsPressed) && (ptKey->usReleaseCount >= ptKey->usReleaseThreshold))
        {
            eFinal = (eTrigger_Release == eKeyTriggerMode) ? eKeyStickyPress : eKeyIdle;
            eEvt   = (eTrigger_Release == eKeyTriggerMode) ? eLongToSticky : eAnyToIdle;

            KeyManager_Transit(ptKey, eKeyIdle, &KeyManager_IdleState, eEvt);
            ptKey->eState = eFinal;
        }
        else
        {
            /* Do Anything */
        }
    }
}

/*==============================================================================
=======                    Function Implement List                       =======
==============================================================================*/
void KeyManager_Init(KeyGroup_t *ptGroup)
{
    uint16_t usIndex      = 0U;
    KeyButton_t *ptButton = NULL;

    if ((NULL != ptGroup) && (NULL != ptGroup->ptKeys))
    {
        for (usIndex = 0U; usIndex < ptGroup->usKeyNum; usIndex++)
        {
            ptButton = &ptGroup->ptKeys[usIndex];

            ptButton->eState         = eKeyIdle;
            ptButton->eStatusEvent   = eKeep;
            ptButton->StateHandler   = &KeyManager_IdleState;
            ptButton->usPressCount   = 0U;
            ptButton->usReleaseCount = 0U;

        }
    }
}

int16_t KeyManager_Scan(KeyGroup_t *ptGroup)
{
    KeyButton_t *ptKey       = NULL;
    uint16_t    usIndex      = 0U;
    int16_t     sWinnerIndex = -1;
    bool        bIsPressed   = false;

    if ((NULL != ptGroup) && (NULL != ptGroup->ptKeys))
    {
        for (usIndex = 0U; usIndex < ptGroup->usKeyNum; usIndex++)
        {
            ptKey = &ptGroup->ptKeys[usIndex];

            if ((ptKey->eState > eKeyPressed) && (ptKey->usPressCount == 0U))
            {
                ptKey->eState = eKeyIdle;
            }

            bIsPressed = KeyManager_IsPressed(ptKey);
            
            KeyManager_UpdateTimers(ptKey, bIsPressed);
            ptKey->eStatusEvent = eKeep;
            
            if (NULL != ptKey->StateHandler)
            {
                ptKey->StateHandler(ptKey, bIsPressed, ptGroup->eTriggerMode);
            }

            /* 只有在消抖通过产生 eIdleToShort 时才判定为互斥赢家 */
            if ((true == ptGroup->bIsExclusive) && (eIdleToShort == ptKey->eStatusEvent))
            {
                sWinnerIndex = (int16_t)usIndex;
                break;
            }
        }

        /* 互斥处理阶段 */
        if (-1 != sWinnerIndex)
        {
            for (usIndex = 0U; usIndex < ptGroup->usKeyNum; usIndex++)
            {
                if (usIndex != (uint16_t)sWinnerIndex)
                {
                    KeyManager_ResetKey(&ptGroup->ptKeys[usIndex]);
                }
            }
        }
    }
    return sWinnerIndex;
}

KeyStatus_e KeyManager_GetStatus(const KeyGroup_t *ptGroup, uint32_t usKeyIndex)
{
    KeyStatus_e eResult = eKeyIdle;

    if ((NULL != ptGroup) && (usKeyIndex < ptGroup->usKeyNum) && (NULL != ptGroup->ptKeys))
    {
        eResult = ptGroup->ptKeys[usKeyIndex].eState;
    }

    return eResult;
}

static void PowerManager_OverState(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal <= pMonitor->usOverRecoverVal)
    {
        pMonitor->eState = VOLTAGE_OVER_RECOVERING;
        pMonitor->ulCounter = 0;
    }
}

static void PowerManager_NormalState(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal > pMonitor->usOverVoltageVal)
    {
        pMonitor->eState = VOLTAGE_OVER_DETECTING;
        pMonitor->ulCounter = 0;
    } else if (usCurrentVal < pMonitor->usUnderVoltageVal)
    {
        pMonitor->eState = VOLTAGE_UNDER_DETECTING;
        pMonitor->ulCounter = 0;
    }
}

static void PowerManager_UnderState(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal >= pMonitor->usUnderRecoverVal)
    {
        pMonitor->eState = VOLTAGE_UNDER_RECOVERING;
        pMonitor->ulCounter = 0;
    }
}

static void PowerManager_OverVoltageCheck(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal > pMonitor->usOverVoltageVal)
    {
        pMonitor->ulCounter++;
        if (pMonitor->ulCounter >= pMonitor->usOverTimeout)
        {
            pMonitor->eState = VOLTAGE_OVER;
            pMonitor->ulCounter = 0;
        }
    }
    else if(usCurrentVal < pMonitor->usUnderVoltageVal)
    {
        pMonitor->eState = VOLTAGE_UNDER_DETECTING;
        pMonitor->ulCounter = 1;
    }
    else
    {
        pMonitor->eState = VOLTAGE_NORMAL;
    }
}

static void PowerManager_UnderVoltageCheck(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal < pMonitor->usUnderVoltageVal)
    {
        pMonitor->ulCounter++;
        if (pMonitor->ulCounter >= pMonitor->usUnderTimeout)
        {
            pMonitor->eState = VOLTAGE_UNDER;
            pMonitor->ulCounter = 0;
        }
    }
    else if(usCurrentVal > pMonitor->usOverVoltageVal)
    {
        pMonitor->eState = VOLTAGE_OVER_DETECTING;
        pMonitor->ulCounter = 1;
    }
    else
    {
        pMonitor->eState = VOLTAGE_NORMAL;
    }
}

static void PowerManager_OverVoltageRecovery(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal <= pMonitor->usOverRecoverVal)
    {
        pMonitor->ulCounter++;
        if (pMonitor->ulCounter >= pMonitor->usOverRecoverTimeout)
        {
            pMonitor->eState = VOLTAGE_NORMAL;
            pMonitor->ulCounter = 0;
        }
    }
    else
    {
        pMonitor->eState = VOLTAGE_OVER;
    }
}

static void PowerManager_UnderVoltageRecovery(VoltageMonitor_t *pMonitor, uint16_t usCurrentVal)
{
    if (usCurrentVal >= pMonitor->usUnderRecoverVal)
    {
        pMonitor->ulCounter++;
        if (pMonitor->ulCounter >= pMonitor->usUnderRecoverTimeout)
        {
            pMonitor->eState    = VOLTAGE_NORMAL;
            pMonitor->ulCounter = 0;
        }
    }
    else
    {
        pMonitor->eState = VOLTAGE_UNDER;
    }
}

static VoltageStateFunc_t g_pfnStateTable[VOLTAGE_STATE_MAX] = 
{
    [VOLTAGE_NORMAL]           = PowerManger_NormalState,
    [VOLTAGE_UNDER_DETECTING]  = PowerManger_UnderVoltageCheck,
    [VOLTAGE_OVER_DETECTING]   = PowerManger_OverVoltageCheck,
    [VOLTAGE_UNDER]            = PowerManger_UnderState,
    [VOLTAGE_UNDER_RECOVERING] = PowerManger_UnderVoltageRecovery,
    [VOLTAGE_OVER]             = PowerManger_OverState,
    [VOLTAGE_OVER_RECOVERING]  = PowerManger_OverVoltageRecovery,
};

/*==============================================================================
=======                    Function Implement List                       =======
==============================================================================*/
void PowerManager_VoltageMonitorInit(VoltageMonitor_t *pMonitorTable, uint8_t ucNumber)
{
    if ((NULL == pMonitorTable) || (0 == ucNumber))
    {
        return;
    }

    m_pVoltageMonitorTable = pMonitorTable;
    m_ucTableNumber        = ucNumber;

    for (uint8_t i = 0; i < m_ucTableNumber; i++)
    {
        m_pVoltageMonitorTable[i].eState    = VOLTAGE_NORMAL;
        m_pVoltageMonitorTable[i].ulCounter = 0;
    }
}

void PowerManager_VoltageMonitorScan()
{
    if ((NULL == m_pVoltageMonitorTable) || (0 == m_ucTableNumber))
    {
        return;
    }

    for (uint8_t i = 0; i < m_ucTableNumber; i++)
    {
        VoltageMonitor_t *pMonitor    = &m_pVoltageMonitorTable[i];
        uint16_t         usCurrentVal = pMonitor->pReadVoltageFunc();

        // 为了后面过Tessy，所以避免使用switch语句导致圈复杂度过高
        if ((pMonitor->eState < VOLTAGE_STATE_MAX) && (NULL != g_pfnStateTable[pMonitor->eState]))
        {
            g_pfnStateTable[pMonitor->eState](pMonitor, usCurrentVal);
        }
    }
}

        return;
    }

    if (eBspWdt_TypeIndependent == m_ptActiveWdtCfg->eType) 
    {
        FL_IWDT_ReloadCounter(IWDT);
    }
    else
    {
        FL_WWDT_ReloadCounter(WWDT);
    }
}

/* 中断服务函数 */

void WDT_IRQHandler(void)
{
    if(FL_WWDT_IsActiveFlag_NearOverflow(WWDT))
    {
        FL_WWDT_ClearFlag_NearOverflow(WWDT);
        FL_WWDT_ReloadCounter(WWDT);
    }

    /* 若程序进入此分支，则表示IWDT中断，请自行处理 */
}
